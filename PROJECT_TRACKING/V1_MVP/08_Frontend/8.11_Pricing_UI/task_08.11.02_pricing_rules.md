## Conditions Schema

```typescript
// Flexible conditions structure stored as JSONB
interface RuleConditions {
  // Product targeting
  products?: string[];           // Specific product IDs
  variants?: string[];           // Specific variant IDs
  categories?: string[];         // Category IDs
  excludeProducts?: string[];    // Excluded product IDs
  
  // Quantity conditions
  minQuantity?: number;          // Minimum items in cart
  maxQuantity?: number;          // Maximum items
  
  // Amount conditions
  minOrderAmount?: number;       // Minimum order value
  maxOrderAmount?: number;       // Maximum order value
  
  // Customer targeting
  customerIds?: string[];        // Specific customers
  customerGroups?: string[];     // Customer groups (wholesale, vip, etc.)
  
  // Time conditions
  weekdays?: number[];           // 0=Sunday, 1=Monday, etc.
  timeRange?: {
    start: string;               // "09:00"
    end: string;                 // "17:00"
  };
  
  // Special conditions
  firstOrderOnly?: boolean;      // New customer first order
  newCustomerDays?: number;      // Customer registered within N days
}
```
